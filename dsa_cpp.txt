    3.2. Reverse Elements (Modify Array)
    ----
    void Reverse(int arr[], int n)
    {
        int i = 0;
        int j = n - 1;
        int temp;
        while (i < j)
        {
            temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
            i++;
            j--;
        }

        cout << "The reversed array elements are: [ ";

        for (int i = 0; i < n; i++)
        {
            cout << arr[i] << " ";
        }

        cout << "]" << endl;
    }

--------- 4. ARRAY ANALYSIS OPERATIONS ---------

    4.1. Find Duplicates
    ----
    int Duplicates(int arr[], int n)
    {
        for (int i = 0; i < n; i++)
        {
            for(int j = i + 1; j < n; j++)
            {
                if (arr[i] == arr[j])
                    return arr[i];
            }
            return -1;
        }
    }

        }

    }


    4.3. Calculate Average
    ----
    int Average(int arr[], int n)
    {
        int sum = 0;
        for(int i = 0; i < n; i++)
        {
            sum = sum + arr[i];
        }

        int avg = sum / n;
        return avg;
    }

--------- 1. LINEAR SEARCH ---------

    1.1. Linear Search (Iterative)
    ----
    int Linear(int a[], int n, int number)
    {
        for (int i = 0; i < n; i++)
        {
            if (a[i] == number)
                return i;
        }
        return -1;
    }


    1.2. Linear Search (Recursive)
    ----
    int LinearRecursive(int a[], int index, int number)
    {
        if (index < 0)
            return -1;

        if (a[index] == number)
            return index;

        else
            return LinearRecursive(a, index - 1, number);
    }


--------- 2. BINARY SEARCH ---------

    Note: Binary search algorithm works on sorted arrays only

    2.1. Binary Search (Iterative)
    ----
    void Binary(int a[], int n, int number)
    {
        int first = 0;
        int last = n - 1;
        int mid;

        while (first <= last)
        {
            mid = (first + last) / 2;

            if (number < a[mid])
            {
                last = mid - 1;
            }

            else if (number > a[mid])
            {
                first = mid + 1;
            }

            else if (number == a[mid])
            {
                cout << "Element found at index : " << mid << endl;
                break;
            }

            if (first > last)
                cout << "Element not found!!!" << endl;
        }
    }


    2.2. Binary Search (Recursive)
    ----
    int BinaryRecursive(int a[], int first, int last, int number)
    {
        if (first > last)
            return -1;

        int mid = (first + last) / 2;

        if (number == a[mid])
            return mid;

        else if (number < a[mid])
            return BinaryRecursive(a, first, mid - 1, number);

        else    // (number > a[mid])
            return BinaryRecursive(a, mid + 1, last, number);
    }

--------- 1. BUBBLE SORT ---------

    void Bubble(int arr[], int n)
    {
        int i, j;
        for (i = 0; i < n - 1; i++)
        {
            for (j = 0; j < n - i - 1; j++)
            {
                if (arr[j] > arr[j + 1])
                {
                    int temp = arr[j + 1];
                    arr[j + 1] = arr[j];
                    arr[j] = temp;
                }
            }
        }
    }


--------- 2. INSERTION SORT ---------

    void Insertion(int arr[], int n)
    {
        int i, j, key;
        for (i = 1; i < n; i++)
        {
            key = arr[i];
            j = i - 1;
            while (j >= 0 && arr[j] > key)
            {
                arr[j + 1] = arr[j];
                j = j - 1;
            }
            arr[j + 1] = key;
        }
    }


--------- 3. SELECTION SORT ---------

    void Swap(int *a, int* b)
    {
        int temp = *a;
        *a = *b;
        *b = temp;
    }


    void Selection(int arr[], int n)
    {
        int i, j, min_idx;
        for (i = 0; i < n - 1; i++)
        {
            min_idx = i;

            for(j = i + 1; j < n; j++)
            {
                if (arr[j] < arr[min_idx])
                    min_idx = j;
            }

            Swap(&arr[min_idx], &arr[i]);
        }
    }


--------- 4. MERGE SORT ---------

    void Merge(int arr[], int low, int mid, int high)
    {
        int i, j, k;

        int nL = mid - low + 1;
        int nR = high - mid;

        int left[nL], right[nR];

        for(i = 0; i < nL; i++)
        {
            left[i] = arr[low + i];
        }

        for(j = 0; j < nR; j++)
        {
            right[j] = arr[mid + 1 + j];
        }

        i = 0;
        j = 0;
        k = low;

        while (i < nL && j < nR)
        {
            if (left[i] <= right[j])
            {
                arr[k] = left[i];
                i++;
            }

            else
            {
                arr[k] = right[j];
                j++;
            }

            k++;
        }

        while (i < nL)
        {
            arr[k] = left[i];
            i++;
            k++;
        }

        while (j < nR)
        {
            arr[k] = right[j];
            j++;
            k++;
        }
    }


    void MergeSort(int arr[], int low, int high)
    {
        int mid;

        if (low < high)
        {
            mid = (low + high) / 2;
            MergeSort(arr, low, mid);
            MergeSort(arr, mid + 1, high);
            Merge(arr, low, mid, high);
        }
    }


--------- 5. QUICK SORT ---------

    void swap(int* a, int* b)
    {
        int t = *a;
        *a = *b;
        *b = t;
    }


    int Partition(int arr[], int first, int last)
    {
        int pivot = arr[last];
        int j = first;

        for(int i = first; i < last; i++)
        {
            if (arr[i] <= pivot)
            {
                swap(&arr[i], &arr[j]);
                j++;
            }
        }

        swap(&arr[j], &arr[last]);
        return j;
    }


    void QuickSort(int arr[], int first, int last)
    {
        int pIndex;

        if (first < last)
        {
            pIndex = Partition(arr, first, last);
            QuickSort(arr, first, pIndex - 1);
            QuickSort(arr, pIndex + 1, last);
        }
    }

            left[i] = arr[low + i];
        }

        for(j = 0; j < nR; j++)
        {
            right[j] = arr[mid + 1 + j];
        }

        i = 0;
        j = 0;
        k = low;

        while (i < nL && j < nR)
        {
            if (left[i] <= right[j])
            {
                arr[k] = left[i];
                i++;
            }

            else
            {
                arr[k] = right[j];
                j++;
            }

            k++;
        }

        while (i < nL)
        {
            arr[k] = left[i];
            i++;
            k++;
        }

        while (j < nR)
        {
            arr[k] = right[j];
            j++;
            k++;
        }
    }


    void MergeSort(int arr[], int low, int high)
    {
        int mid;

        if (low < high)
        {
            mid = (low + high) / 2;
            MergeSort(arr, low, mid);
            MergeSort(arr, mid + 1, high);
            Merge(arr, low, mid, high);
        }
    }


    void swap(int* a, int* b)
    {
        int t = *a;
        *a = *b;
        *b = t;
    }


    int Partition(int arr[], int first, int last)
    {
        int pivot = arr[last];
        int j = first;

        for(int i = first; i < last; i++)
        {
            if (arr[i] <= pivot)
            {
                swap(&arr[i], &arr[j]);
                j++;
            }
        }

        swap(&arr[j], &arr[last]);
        return j;
    }


    void QuickSort(int arr[], int first, int last)
    {
        int pIndex;

        if (first < last)
        {
            pIndex = Partition(arr, first, last);
            QuickSort(arr, first, pIndex - 1);
            QuickSort(arr, pIndex + 1, last);
        }
    }

        for (int i = 0; i < n; i++)
        {
            cout << arr[i] << " ";
        }

        cout << "]" << endl;
    }

    ----
    void display()
    {
        if (front == rear)
            cout << "Queue is empty. " << endl;
        else
        {
            cout << "Queue elements are : ";
            for (int i = front; i < rear; i++)
            {
                cout << queue[i] << " ";
            }
            cout << endl;
        }
    }


    4.2. Is Empty Check
    ----
    bool isEmpty()
    {
        if (front == rear)
            return true;
        else
            return false;
    }


    4.3. Is Full Check
    ----
    bool isFull()
    {
        if (rear == CAPACITY)
            return true;
        else
            return false;
    }


    4.4. Get Front Element
    ----
    int frontelement()
    {
        return queue[front];
    }


    4.5. Get Back Element
    ----
    int backelement()
    {
        return queue[rear - 1];
    }
